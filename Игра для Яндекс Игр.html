game.js

javascript
let score = 0;
let pointsPerClick = 1;

const scoreDisplay = document.getElementById('score');
const clickButton = document.getElementById('click');
const upgradeButton = document.getElementById('upgrade');

clickButton.onclick = () => {
  score += pointsPerClick;
  scoreDisplay.textContent = score;
};

upgradeButton.onclick = () => {
  if (score >= 10) {
    score -= 10;
    pointsPerClick++;
    upgradeButton.textContent = `Улучшить (+${pointsPerClick} за клик) — 10 очков`;
    scoreDisplay.textContent = score;
  }
};

Как играть:

  1. Жмёшь "Клик!" — получаешь очки.

  2. Накопил 10 очков — жмёшь "Улучшить" → каждый клик даёт больше очков.

  1. Создай архив с index.html, game.js.

  2. Зарегистрируйся на https://yandex.ru/dev/games/ и загрузи.

game.js

javascript
let score = parseInt(localStorage.getItem("score")) || 0;
let best = parseInt(localStorage.getItem("best")) || 0;
let pointsPerClick = parseInt(localStorage.getItem("pointsPerClick")) || 1;
let autoClickers = parseInt(localStorage.getItem("autoClickers")) || 0;

const scoreDisplay = document.getElementById('score');
const bestDisplay = document.getElementById('best');
const clickBtn = document.getElementById('clickBtn');
const upgradeClick = document.getElementById('upgradeClick');
const buyAuto = document.getElementById('buyAuto');
const resetGame = document.getElementById('resetGame');

function update() {
  scoreDisplay.textContent = score;
  bestDisplay.textContent = best;
  localStorage.setItem("score", score);
  localStorage.setItem("best", best);
  localStorage.setItem("pointsPerClick", pointsPerClick);
  localStorage.setItem("autoClickers", autoClickers);
}

clickBtn.onclick = () => {
  score += pointsPerClick;
  if (score > best) best = score;
  update();
};

upgradeClick.onclick = () => {
  if (score >= 10) {
    score -= 10;
    pointsPerClick++;
    update();
  }
};

buyAuto.onclick = () => {
  if (score >= 50) {
    score -= 50;
    autoClickers++;
    update();
  }
};

resetGame.onclick = () => {
  if (confirm("Точно сбросить игру?")) {
    score = 0; pointsPerClick = 1; autoClickers = 0; best = 0;
    update();
  }
};

// Автоклик каждые 1 сек
setInterval(() => {
  if (autoClickers > 0) {
    score += autoClickers;
    if (score > best) best = score;
    update();
  }
}, 1000);

update();
